# Phase 1: Earth Globe — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Earth Globe view — a geographic visualization of electronic music scenes using globe.gl, with city markers, distribution arcs, genre heatmap, fly-to interactions, and a minimap. This is a separate view mode from the existing Genre World, switchable via Zustand `viewMode` state.

**Architecture:** globe.gl (vanilla Three.js) integrated into React via ref-based approach. NOT React Three Fiber — globe.gl manages its own Three.js scene. The existing R3F genre world stays untouched. A top-level view switch conditionally renders either the R3F Canvas or the globe.gl container.

**Tech Stack:** globe.gl v2.45, three (already installed), Zustand (already installed), GSAP (already installed)

**Spec:** `docs/superpowers/specs/2026-03-16-discoworld-v2-mega-design.md` section 3.1

```jsx
// packages/web/src/components/EarthGlobe.jsx
import { useEffect, useRef, useCallback } from 'react'
import Globe from 'globe.gl'
import * as THREE from 'three'
import useStore from '../stores/useStore'
import './EarthGlobe.css'

export default function EarthGlobe() {
  const containerRef = useRef(null)
  const globeRef = useRef(null)

  useEffect(() => {
    if (!containerRef.current || globeRef.current) return

    const globe = Globe()
      .globeImageUrl(null)
      .backgroundColor('rgba(0,0,0,0)')
      .showAtmosphere(false)  // We'll add custom atmosphere
      .showGraticules(true)
      .atmosphereColor('rgba(0,0,0,0)')
      .width(containerRef.current.clientWidth)
      .height(containerRef.current.clientHeight)

    globe(containerRef.current)

    // Access the Three.js scene and renderer
    const scene = globe.scene()
    const renderer = globe.renderer()
    const camera = globe.camera()

    // Dark matte globe material
    const globeMesh = scene.children.find(
      c => c.type === 'Mesh' && c.geometry?.type === 'SphereGeometry'
    )
    if (globeMesh) {
      globeMesh.material = new THREE.MeshPhongMaterial({
        color: 0x0a0a1a,
        transparent: false,
        shininess: 5,
      })
    }

    // Graticule styling
    globe.graticulesDashGap(0.5)
    globe.graticulesColor(() => 'rgba(100, 200, 255, 0.06)')

    // Renderer settings
    renderer.toneMapping = THREE.ACESFilmicToneMapping
    renderer.toneMappingExposure = 1.0

    // Lighting
    const ambient = new THREE.AmbientLight(0x334466, 0.3)
    scene.add(ambient)
    const directional = new THREE.DirectionalLight(0xaaccff, 0.8)
    directional.position.set(5, 3, 5)
    scene.add(directional)

    // Auto-rotate
    globe.controls().autoRotate = true
    globe.controls().autoRotateSpeed = 0.3
    globe.controls().enableDamping = true
    globe.controls().dampingFactor = 0.1

    globeRef.current = globe

    // Resize handler
    const onResize = () => {
      if (!containerRef.current) return
      globe.width(containerRef.current.clientWidth)
      globe.height(containerRef.current.clientHeight)
    }
    window.addEventListener('resize', onResize)

    return () => {
      window.removeEventListener('resize', onResize)
      globe._destructor?.()
      globeRef.current = null
    }
  }, [])

  return (
    <div className="earth-globe-container" ref={containerRef} />
  )
}
```

```js
// packages/web/src/components/GlobeAtmosphere.js
/**
 * Premium atmosphere: Fresnel backface shader + double glow spheres.
 * Adds to the globe.gl Three.js scene directly.
 */
import * as THREE from 'three'

const FRESNEL_VERTEX = `
  varying vec3 vNormal;
  varying vec3 vPosition;
  void main() {
    vNormal = normalize(normalMatrix * normal);
    vPosition = (modelViewMatrix * vec4(position, 1.0)).xyz;
    gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
  }
`

const FRESNEL_FRAGMENT = `
  varying vec3 vNormal;
  varying vec3 vPosition;
  uniform vec3 glowColor;
  uniform float intensity;
  uniform float power;
  void main() {
    vec3 viewDir = normalize(-vPosition);
    float fresnel = pow(1.0 - abs(dot(vNormal, viewDir)), power);
    gl_FragColor = vec4(glowColor, fresnel * intensity);
  }
`

/**
 * @param {THREE.Scene} scene
 * @param {number} globeRadius - globe.gl default is 100
 * @returns {Function} cleanup function
 */
export function addAtmosphere(scene, globeRadius = 100) {
  const meshes = []

  // 1. Fresnel atmosphere shell (BackSide)
  const atmosGeom = new THREE.SphereGeometry(globeRadius * 1.015, 64, 64)
  const atmosMat = new THREE.ShaderMaterial({
    vertexShader: FRESNEL_VERTEX,
    fragmentShader: FRESNEL_FRAGMENT,
    uniforms: {
      glowColor: { value: new THREE.Color(0x00ddff) },  // cyan
      intensity: { value: 0.7 },
      power: { value: 3.5 },
    },
    side: THREE.BackSide,
    transparent: true,
    depthWrite: false,
    blending: THREE.AdditiveBlending,
  })
  const atmosMesh = new THREE.Mesh(atmosGeom, atmosMat)
  scene.add(atmosMesh)
  meshes.push(atmosMesh)

  // 2. Outer glow sphere (larger, subtler)
  const outerGeom = new THREE.SphereGeometry(globeRadius * 1.08, 48, 48)
  const outerMat = new THREE.MeshBasicMaterial({
    color: 0x0088cc,
    transparent: true,
    opacity: 0.08,
    side: THREE.BackSide,
    depthWrite: false,
    blending: THREE.AdditiveBlending,
  })
  const outerMesh = new THREE.Mesh(outerGeom, outerMat)
  scene.add(outerMesh)
  meshes.push(outerMesh)

  // 3. Inner glow sphere (tighter, brighter)
  const innerGeom = new THREE.SphereGeometry(globeRadius * 1.03, 48, 48)
  const innerMat = new THREE.MeshBasicMaterial({
    color: 0x00ccff,
    transparent: true,
    opacity: 0.05,
    side: THREE.BackSide,
    depthWrite: false,
    blending: THREE.AdditiveBlending,
  })
  const innerMesh = new THREE.Mesh(innerGeom, innerMat)
  scene.add(innerMesh)
  meshes.push(innerMesh)

  return () => {
    meshes.forEach(m => {
      scene.remove(m)
      m.geometry.dispose()
      m.material.dispose()
    })
  }
}
```

```js
// packages/web/src/components/GlobeStarfield.js
/**
 * 2K-point starfield using InstancedMesh for single draw call.
 */
import * as THREE from 'three'

/**
 * @param {THREE.Scene} scene
 * @param {number} count - number of stars
 * @param {number} radius - sphere radius to place stars on
 * @returns {Function} cleanup
 */
export function addStarfield(scene, count = 2000, radius = 800) {
  const geometry = new THREE.SphereGeometry(0.3, 4, 4)
  const material = new THREE.MeshBasicMaterial({
    color: 0xffffff,
    transparent: true,
    opacity: 0.8,
  })

  const mesh = new THREE.InstancedMesh(geometry, material, count)
  const dummy = new THREE.Object3D()
  const color = new THREE.Color()

  for (let i = 0; i < count; i++) {
    // Random point on sphere
    const theta = Math.random() * Math.PI * 2
    const phi = Math.acos(2 * Math.random() - 1)
    const r = radius + (Math.random() - 0.5) * 200

    dummy.position.set(
      r * Math.sin(phi) * Math.cos(theta),
      r * Math.sin(phi) * Math.sin(theta),
      r * Math.cos(phi)
    )
    // Random size variation
    const scale = 0.3 + Math.random() * 1.2
    dummy.scale.setScalar(scale)
    dummy.updateMatrix()
    mesh.setMatrixAt(i, dummy.matrix)

    // Slight color variation (white to pale blue)
    const hue = 0.55 + Math.random() * 0.1  // blue range
    const sat = Math.random() * 0.3
    const lum = 0.7 + Math.random() * 0.3
    color.setHSL(hue, sat, lum)
    mesh.setColorAt(i, color)
  }

  mesh.instanceMatrix.needsUpdate = true
  mesh.instanceColor.needsUpdate = true
  scene.add(mesh)

  return () => {
    scene.remove(mesh)
    geometry.dispose()
    material.dispose()
  }
}
```

```js
// packages/web/src/components/GlobeBloom.js
/**
 * UnrealBloomPass at half resolution for premium glow.
 * Applied to globe.gl's renderer directly.
 */
import * as THREE from 'three'
import { EffectComposer } from 'three/examples/jsm/postprocessing/EffectComposer.js'
import { RenderPass } from 'three/examples/jsm/postprocessing/RenderPass.js'
import { UnrealBloomPass } from 'three/examples/jsm/postprocessing/UnrealBloomPass.js'
import { OutputPass } from 'three/examples/jsm/postprocessing/OutputPass.js'

/**
 * @param {THREE.WebGLRenderer} renderer
 * @param {THREE.Scene} scene
 * @param {THREE.Camera} camera
 * @returns {{ composer: EffectComposer, resize: Function, cleanup: Function }}
 */
export function setupBloom(renderer, scene, camera) {
  const size = renderer.getSize(new THREE.Vector2())

  // Half-res for performance
  const composer = new EffectComposer(renderer)
  composer.setSize(size.x, size.y)
  composer.setPixelRatio(Math.min(window.devicePixelRatio, 1.5))

  const renderPass = new RenderPass(scene, camera)
  composer.addPass(renderPass)

  const bloomPass = new UnrealBloomPass(
    new THREE.Vector2(size.x / 2, size.y / 2),
    1.2,    // strength
    0.4,    // radius
    0.8     // threshold
  )
  composer.addPass(bloomPass)

  const outputPass = new OutputPass()
  composer.addPass(outputPass)

  const resize = (width, height) => {
    composer.setSize(width, height)
    bloomPass.resolution.set(width / 2, height / 2)
  }

  const cleanup = () => {
    composer.dispose()
  }

  return { composer, resize, cleanup }
}
```

In `EarthGlobe.jsx`, after setting up the globe, hook into the render loop:

```jsx
import { setupBloom } from './GlobeBloom'

// Inside useEffect, after globe init:
const { composer, resize: resizeBloom, cleanup: cleanupBloom } = setupBloom(
  renderer, scene, camera
)

// Override globe.gl's render loop to use composer
const originalAnimate = renderer.info  // globe.gl handles its own RAF
// We need to hook into the postrender callback
globe.onGlobeReady(() => {
  // Replace the default render with bloom composer
  globe.postProcessingComposer(composer)
})

// If postProcessingComposer is not available in globe.gl v2.45,
// use the alternative approach — monkey-patch the animation loop:
let animFrameId = null
const animate = () => {
  animFrameId = requestAnimationFrame(animate)
  globe.controls().update()
  composer.render()
}
// Disable globe.gl's internal rendering and use ours
renderer.setAnimationLoop(null)
animate()

// Update resize handler to include bloom
const onResize = () => {
  if (!containerRef.current) return
  const w = containerRef.current.clientWidth
  const h = containerRef.current.clientHeight
  globe.width(w).height(h)
  resizeBloom(w, h)
}
```

**Note for implementer:** globe.gl v2.45 manages its own animation loop. The cleanest approach is:
1. Call `renderer.setAnimationLoop(null)` to stop globe.gl's internal loop
2. Run your own rAF loop that calls `globe.controls().update()` then `composer.render()`
3. This gives full control over post-processing

If this causes issues with globe.gl internals (labels, interactions), fall back to the simpler approach of letting globe.gl render normally and adding a second pass with bloom on top via `renderer.autoClear = false`.

```python
# packages/api/routes/globe.py
"""API routes for Earth Globe view."""
from fastapi import APIRouter, Query
from db import get_db

router = APIRouter(prefix="/api/globe", tags=["globe"])


@router.get("/cities")
def get_cities():
    """Return cities with release counts and top genres."""
    with get_db() as conn:
        # Count releases per country
        country_counts = conn.execute(
            "SELECT country, COUNT(*) as count FROM releases "
            "WHERE country != '' GROUP BY country ORDER BY count DESC"
        ).fetchall()

        # Top styles per country
        country_styles = {}
        for row in country_counts[:30]:
            country = row["country"]
            styles = conn.execute(
                "SELECT styles FROM releases WHERE country = ? LIMIT 5000",
                (country,)
            ).fetchall()
            # Aggregate style counts
            import json
            from collections import Counter
            style_counter = Counter()
            for r in styles:
                if r["styles"]:
                    for s in json.loads(r["styles"]):
                        style_counter[s] += 1
            country_styles[country] = style_counter.most_common(5)

        return {
            "countries": [
                {
                    "code": row["country"],
                    "release_count": row["count"],
                    "top_styles": country_styles.get(row["country"], []),
                }
                for row in country_counts[:50]
            ],
        }


@router.get("/arcs")
def get_arcs(limit: int = Query(100, le=500)):
    """Return distribution arcs: label connections between countries."""
    with get_db() as conn:
        arcs = conn.execute(
            "SELECT r1.country as src, r2.country as dst, COUNT(*) as weight "
            "FROM releases r1 "
            "JOIN releases r2 ON r1.label = r2.label AND r1.country != r2.country "
            "WHERE r1.country != '' AND r2.country != '' AND r1.id < r2.id "
            "GROUP BY r1.country, r2.country "
            "ORDER BY weight DESC LIMIT ?",
            (limit,)
        ).fetchall()
        return {"arcs": [dict(a) for a in arcs]}


@router.get("/heatmap")
def get_heatmap():
    """Return country-level release density for choropleth/heatmap."""
    with get_db() as conn:
        rows = conn.execute(
            "SELECT country, COUNT(*) as count FROM releases "
            "WHERE country != '' GROUP BY country ORDER BY count DESC"
        ).fetchall()
        return {"heatmap": [dict(r) for r in rows]}
```

```jsx
function buildArcs(cities) {
  const arcs = []
  for (let i = 0; i < cities.length; i++) {

      const shared = cities[i].genres.filter(g => cities[j].genres.includes(g))
      if (shared.length > 0) {

          startLat: cities[i].lat,
          startLng: cities[i].lng,
          endLat: cities[j].lat,
          endLng: cities[j].lng,

          weight: shared.length,
          shared,
          label: `${cities[i].name} ↔ ${cities[j].name}: ${shared.join(', ')}`,
        })
      }
    }
  }
  return arcs
}
```

```jsx
// packages/web/src/components/Minimap.jsx
import { useState, useEffect, useRef } from 'react'
import useStore from '../stores/useStore'
import { CITIES } from '../data/cities'

/**
 * Minimap: 2D Mercator projection of cities with "you are here" indicator.
 * Clickable — teleports globe camera to clicked location.
 */
export default function Minimap() {
  const viewMode = useStore(s => s.viewMode)
  const selectedCity = useStore(s => s.selectedCity)
  const setSelectedCity = useStore(s => s.setSelectedCity)
  const [viewCenter, setViewCenter] = useState({ lat: 30, lng: 0 })

  if (viewMode !== 'earth') return null

  // Mercator projection: lat/lng to x/y (0-240, 0-120)
  const W = 240
  const H = 120
  const project = (lat, lng) => ({
    x: ((lng + 180) / 360) * W,
    y: ((90 - lat) / 180) * H,
  })

  const handleClick = (city) => {
    setSelectedCity(city)
  }

  return (
    <div className="minimap">
      <svg width={W} height={H} viewBox={`0 0 ${W} ${H}`}>
        {/* Background */}
        <rect width={W} height={H} fill="rgba(5, 5, 20, 0.9)" rx="4" />

        {/* Simple continent outlines (simplified) */}
        <rect x="0" y="0" width={W} height={H} fill="none" stroke="rgba(100,200,255,0.08)" strokeWidth="0.5" />

        {/* Grid lines */}
        {[0, 30, 60, 90, 120, 150, 180, 210, 240].map(x => (
          <line key={`v${x}`} x1={x} y1={0} x2={x} y2={H} stroke="rgba(100,200,255,0.04)" strokeWidth="0.5" />
        ))}
        {[0, 30, 60, 90, 120].map(y => (
          <line key={`h${y}`} x1={0} y1={y} x2={W} y2={y} stroke="rgba(100,200,255,0.04)" strokeWidth="0.5" />
        ))}

        {/* City dots */}
        {CITIES.map(city => {
          const p = project(city.lat, city.lng)
          const isSelected = selectedCity?.name === city.name
          const r = Math.max(2, city.genres.length * 0.8)
          return (
            <g key={city.name} onClick={() => handleClick(city)} style={{ cursor: 'pointer' }}>
              <circle
                cx={p.x}
                cy={p.y}
                r={r}
                fill={isSelected ? '#ffbb33' : '#ffbb33'}
                opacity={isSelected ? 1 : 0.5}
              />
              {isSelected && (
                <circle
                  cx={p.x}
                  cy={p.y}
                  r={r + 4}
                  fill="none"
                  stroke="#ffbb33"
                  strokeWidth="1"
                  opacity="0.5"
                >
                  <animate attributeName="r" from={r + 2} to={r + 8} dur="1.5s" repeatCount="indefinite" />
                  <animate attributeName="opacity" from="0.6" to="0" dur="1.5s" repeatCount="indefinite" />
                </circle>
              )}
            </g>
          )
        })}

        {/* "You are here" dot — tracks globe camera center */}
        {viewCenter && (() => {
          const p = project(viewCenter.lat, viewCenter.lng)
          return (
            <circle
              cx={p.x}
              cy={p.y}
              r="3"
              fill="#66ccff"
              stroke="#fff"
              strokeWidth="0.5"
            />
          )
        })()}
      </svg>
    </div>
  )
}
```
